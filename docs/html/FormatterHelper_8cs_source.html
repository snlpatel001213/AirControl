41  public static bool ReplaceContent(ref string content, int dataIndex, string numericFormatter, Serie serie,
42  BaseChart chart, DataZoom dataZoom = null)
43  {
44  var foundDot = false;
45  var mc = s_Regex.Matches(content);
46  foreach (var m in mc)
47  {
48  var old = m.ToString();
49  var args = s_RegexSub.Matches(m.ToString());
50  var argsCount = args.Count;
51  if (argsCount <= 0) continue;
52  int targetIndex = 0;
53  char p = GetSerieIndex(args[0].ToString(), ref targetIndex);
54  if (targetIndex >= 0)
55  {
56  serie = chart.series.GetSerie(targetIndex);
57  if (serie == null) continue;
58  }
59  else if (serie != null)
60  {
61  targetIndex = serie.index;
62  }
63  else
64  {
65  serie = chart.series.GetSerie(0);
66  targetIndex = 0;
67  }
68  if (serie == null) continue;
69  if (p == '.')
70  {
71  var bIndex = targetIndex;
72  if (argsCount >= 2)
73  {
74  var args1Str = args[1].ToString();
75  if (s_RegexN.IsMatch(args1Str)) bIndex = int.Parse(args1Str);
76  }
77  content = content.Replace(old, ChartCached.ColorToDotStr(chart.theme.GetColor(bIndex)));
78  foundDot = true;
79  }
80  else if (p == 'a' || p == 'A')
81  {
82  if (argsCount == 1)
83  {
84  content = content.Replace(old, serie.name);
85  }
86  }
87  else if (p == 'b' || p == 'B' || p == 'e' || p == 'E')
88  {
89  var bIndex = dataIndex;
90  if (argsCount >= 2)
91  {
92  var args1Str = args[1].ToString();
93  if (s_RegexN.IsMatch(args1Str))
94  bIndex = int.Parse(args1Str);
95  }
96  var needCategory = (p != 'e' && p != 'E') && (serie.type == SerieType.Line || serie.type == SerieType.Bar);
97  if (needCategory)
98  {
99  var category = (chart as CoordinateChart).GetTooltipCategory(dataIndex, serie, dataZoom);
100  content = content.Replace(old, category);
101  }
102  else
103  {
104  var serieData = serie.GetSerieData(bIndex, dataZoom);
105  content = content.Replace(old, serieData.name);
106  }
107  }
108  else if (p == 'c' || p == 'C' || p == 'd' || p == 'D')
109  {
110  var isPercent = p == 'd' || p == 'D';
111  var bIndex = dataIndex;
112  var dimensionIndex = -1;
113  if (argsCount >= 2)
114  {
115  var args1Str = args[1].ToString();
116  if (s_RegexFn.IsMatch(args1Str))
117  {
118  numericFormatter = args1Str;
119  }
120  else if (s_RegexN_N.IsMatch(args1Str))
121  {
122  var temp = args1Str.Split('-');
123  bIndex = int.Parse(temp[0]);
124  dimensionIndex = int.Parse(temp[1]);
125  }
126  else if (s_RegexN.IsMatch(args1Str))
127  {
128  dimensionIndex = int.Parse(args1Str);
129  }
130  else
131  {
132  Debug.LogError("unmatch:" + args1Str);
133  continue;
134  }
135  }
136  if (argsCount >= 3)
137  {
138  numericFormatter = args[2].ToString();
139  }
140  if (dimensionIndex == -1) dimensionIndex = 1;
141  if (numericFormatter == string.Empty)
142  {
143  numericFormatter = SerieHelper.GetNumericFormatter(serie, serie.GetSerieData(bIndex));
144  }
145  var value = serie.GetData(bIndex, dimensionIndex, dataZoom);
146  if (isPercent)
147  {
148  var total = serie.GetDataTotal(dimensionIndex);
149  var percent = total == 0 ? 0 : value / serie.yTotal * 100;
150  content = content.Replace(old, ChartCached.FloatToStr(percent, numericFormatter));
151  }
152  else
153  {
154  content = content.Replace(old, ChartCached.FloatToStr(value, numericFormatter));
155  }
156  }
157  }
158  content = s_RegexNewLine.Replace(content, PH_NN);
159  return foundDot;
160  }

217  public static void ReplaceSerieLabelContent(ref string content, string numericFormatter, double value, double total,
218  string serieName, string dataName, Color color)
219  {
220  var mc = s_RegexForSerieLabel.Matches(content);
221  foreach (var m in mc)
222  {
223  var old = m.ToString();
224  var args = s_RegexSubForSerieLabel.Matches(old);
225  var argsCount = args.Count;
226  if (argsCount <= 0) continue;
227  var p = args[0].ToString().ElementAt(0);
228  if (argsCount >= 2)
229  {
230  numericFormatter = args[1].ToString();
231  }
232  if (p == '.')
233  {
234  content = content.Replace(old, ChartCached.ColorToDotStr(color));
235  }
236  else if (p == 'a' || p == 'A')
237  {
238  content = content.Replace(old, serieName);
239  }
240  else if (p == 'b' || p == 'B' || p == 'e' || p == 'E')
241  {
242  content = content.Replace(old, dataName);
243  }
244  else if (p == 'c' || p == 'C' || p == 'd' || p == 'D')
245  {
246  var isPercent = p == 'd' || p == 'D';
247  if (isPercent)
248  {
249  if (total != 0)
250  content = content.Replace(old, ChartCached.FloatToStr(value / total * 100, numericFormatter));
251  }
252  else
253  {
254  content = content.Replace(old, ChartCached.FloatToStr(value, numericFormatter));
255  }
256  }
257  }
258  content = TrimAndReplaceLine(content);
259  }
