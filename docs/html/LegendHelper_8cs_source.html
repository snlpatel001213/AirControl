81  public static void ResetItemPosition(Legend legend, Vector3 chartPos, float chartWidth, float chartHeight)
82  {
83  var startX = 0f;
84  var startY = 0f;

87  UpdateLegendWidthAndHeight(legend, legendMaxWidth, legendMaxHeight);
88  var legendRuntimeWidth = legend.runtimeWidth;
89  var legendRuntimeHeight = legend.runtimeHeight;
90  var isVertical = legend.orient == Orient.Vertical;
91  switch (legend.location.align)
92  {
93  case Location.Align.TopCenter:
94  startX = chartPos.x + chartWidth / 2 - legendRuntimeWidth / 2;
95  startY = chartPos.y + chartHeight - legend.location.top;
96  break;
97  case Location.Align.TopLeft:
98  startX = chartPos.x + legend.location.left;
99  startY = chartPos.y + chartHeight - legend.location.top;
100  break;
101  case Location.Align.TopRight:
102  startX = chartPos.x + chartWidth - legendRuntimeWidth - legend.location.right;
103  startY = chartPos.y + chartHeight - legend.location.top;
104  break;
105  case Location.Align.Center:
106  startX = chartPos.x + chartWidth / 2 - legendRuntimeWidth / 2;
107  startY = chartPos.y + chartHeight / 2 + legendRuntimeHeight / 2;
108  break;
109  case Location.Align.CenterLeft:
110  startX = chartPos.x + legend.location.left;
111  startY = chartPos.y + chartHeight / 2 + legendRuntimeHeight / 2;
112  break;
113  case Location.Align.CenterRight:
114  startX = chartPos.x + chartWidth - legendRuntimeWidth - legend.location.right;
115  startY = chartPos.y + chartHeight / 2 + legendRuntimeHeight / 2;
116  break;
117  case Location.Align.BottomCenter:
118  startX = chartPos.x + chartWidth / 2 - legendRuntimeWidth / 2;
119  startY = chartPos.y + legendRuntimeHeight + legend.location.bottom;
120  break;
121  case Location.Align.BottomLeft:
122  startX = chartPos.x + legend.location.left;
123  startY = chartPos.y + legendRuntimeHeight + legend.location.bottom;
124  break;
125  case Location.Align.BottomRight:
126  startX = chartPos.x + chartWidth - legendRuntimeWidth - legend.location.right;
127  startY = chartPos.y + legendRuntimeHeight + legend.location.bottom;
128  break;
129  }
130  if (isVertical) SetVerticalItemPosition(legend, legendMaxHeight, startX, startY);
131  else SetHorizonalItemPosition(legend, legendMaxWidth, startX, startY);

134  private static void SetVerticalItemPosition(Legend legend, float legendMaxHeight, float startX, float startY)
135  {
136  var currHeight = 0f;
137  var offsetX = 0f;
138  var row = 0;
139  foreach (var kv in legend.buttonList)
140  {
141  var item = kv.Value;

143  {

169  private static void UpdateLegendWidthAndHeight(Legend legend, float maxWidth, float maxHeight)
170  {
171  var width = 0f;
172  var height = 0f;
173  var realHeight = 0f;
174  var realWidth = 0f;
175  legend.runtimeEachWidth.Clear();
176  legend.runtimeEachHeight = 0;
177  if (legend.orient == Orient.Horizonal)
178  {
179  foreach (var kv in legend.buttonList)
180  {
181  if (width + kv.Value.width > maxWidth)
182  {
183  realWidth = width - legend.itemGap;
184  realHeight += height + legend.itemGap;
185  if (legend.runtimeEachHeight < height + legend.itemGap)
186  {
187  legend.runtimeEachHeight = height + legend.itemGap;
188  }
189  height = 0;
190  width = 0;
191  }
192  width += kv.Value.width + legend.itemGap;
193  if (kv.Value.height > height)
194  height = kv.Value.height;
195  }
196  width -= legend.itemGap;
197  legend.runtimeHeight = realHeight + height;
198  legend.runtimeWidth = realWidth > 0 ? realWidth : width;
199  }
200  else
201  {
202  var row = 0;
203  foreach (var kv in legend.buttonList)
204  {
205  if (height + kv.Value.height > maxHeight)
206  {
207  realHeight = height - legend.itemGap;
208  realWidth += width + legend.itemGap;
209  legend.runtimeEachWidth[row] = width + legend.itemGap;
210  row++;
211  height = 0;
212  width = 0;
213  }
214  height += kv.Value.height + legend.itemGap;
215  if (kv.Value.width > width)
216  width = kv.Value.width;
217  }
218  height -= legend.itemGap;
219  legend.runtimeHeight = realHeight > 0 ? realHeight : height;
220  legend.runtimeWidth = realWidth + width;
221  }
222  }
