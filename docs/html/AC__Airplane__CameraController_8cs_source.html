405  public void OnCameraChange(Camera activeCamera)
406  {
407  int targetDisplay = 1;
408  foreach (var pass in capturePasses)
409  {
410  if (pass.camera == activeCamera)
411  continue;
412 
413  // cleanup capturing camera
414  pass.camera.RemoveAllCommandBuffers();
415 
416  // copy all "main" camera parameters into capturing camera
417  pass.camera.CopyFrom(activeCamera);
418 
419  // set targetDisplay here since it gets overriden by CopyFrom()
420  pass.camera.targetDisplay = targetDisplay++;
421  }
422 
423  // cache materials and setup material properties
424  if (!opticalFlowMaterial || opticalFlowMaterial.shader != opticalFlowShader)
425  opticalFlowMaterial = new Material(opticalFlowShader);
426  opticalFlowMaterial.SetFloat("_Sensitivity", opticalFlowSensitivity);
427 
428  // setup command buffers and replacement shaders
429  SetupCameraWithReplacementShader(capturePasses[1].camera, uberReplacementShader, ReplacementMode.ObjectId, Color.black);
430  SetupCameraWithReplacementShader(capturePasses[2].camera, uberReplacementShader, ReplacementMode.CatergoryId, Color.black);
431  SetupCameraWithReplacementShader(capturePasses[3].camera, uberReplacementShader, ReplacementMode.DepthCompressed, Color.white);
432  SetupCameraWithReplacementShader(capturePasses[4].camera, uberReplacementShader, ReplacementMode.Normals, Color.black);
433  SetupCameraWithPostShader(capturePasses[5].camera, opticalFlowMaterial, DepthTextureMode.Depth | DepthTextureMode.MotionVectors);
434  }
