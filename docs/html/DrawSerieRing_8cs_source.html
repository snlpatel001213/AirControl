65  public void DrawSerie(VertexHelper vh, Serie serie)
66  {
67  if (serie.type != SerieType.Ring) return;
68  if (!serie.show || serie.animation.HasFadeOut()) return;
69  var data = serie.data;
70  serie.animation.InitProgress(data.Count, serie.startAngle, serie.startAngle + 360);
71  SerieHelper.UpdateCenter(serie, chart.chartPosition, chart.chartWidth, chart.chartHeight);
72  TitleStyleHelper.CheckTitle(serie, ref chart.m_ReinitTitle, ref m_UpdateTitleText);
73  SerieLabelHelper.CheckLabel(serie, ref chart.m_ReinitLabel, ref m_UpdateLabelText);
74  var dataChangeDuration = serie.animation.GetUpdateAnimationDuration();
75  var ringWidth = serie.runtimeOutsideRadius - serie.runtimeInsideRadius;
76  var dataChanging = false;
77  for (int j = 0; j < data.Count; j++)
78  {
79  var serieData = data[j];
80  if (!serieData.show) continue;
81  if (serieData.IsDataChanged()) dataChanging = true;
82  var value = serieData.GetFirstData(dataChangeDuration);
83  var max = serieData.GetLastData();
84  var degree = (float)(360 * value / max);
85  var startDegree = GetStartAngle(serie);
86  var toDegree = GetToAngle(serie, degree);
87  var itemStyle = SerieHelper.GetItemStyle(serie, serieData, serieData.highlighted);
88  var itemColor = SerieHelper.GetItemColor(serie, serieData, chart.theme, j, serieData.highlighted);
89  var itemToColor = SerieHelper.GetItemToColor(serie, serieData, chart.theme, j, serieData.highlighted);
90  var outsideRadius = serie.runtimeOutsideRadius - j * (ringWidth + serie.ringGap);
91  var insideRadius = outsideRadius - ringWidth;
92  var centerRadius = (outsideRadius + insideRadius) / 2;
93  var borderWidth = itemStyle.borderWidth;
94  var borderColor = itemStyle.borderColor;
95  var roundCap = serie.roundCap && insideRadius > 0;
96 
97  serieData.runtimePieStartAngle = serie.clockwise ? startDegree : toDegree;
98  serieData.runtimePieToAngle = serie.clockwise ? toDegree : startDegree;
99  serieData.runtimePieInsideRadius = insideRadius;
100  serieData.runtimePieOutsideRadius = outsideRadius;
101  if (itemStyle.backgroundColor.a != 0)
102  {
103  UGL.DrawDoughnut(vh, serie.runtimeCenterPos, insideRadius, outsideRadius, itemStyle.backgroundColor,
104  itemStyle.backgroundColor, Color.clear, 0, 360, borderWidth, borderColor, 0,
105  chart.settings.cicleSmoothness, false, serie.clockwise);
106  }
107  var isGradient = !UGLHelper.IsValueEqualsColor(itemColor, itemToColor);
108  if (isGradient)
109  {
110  if (serie.clockwise)
111  itemToColor = Color.Lerp(itemColor, itemToColor, toDegree / (startDegree + 360));
112  else
113  itemToColor = Color.Lerp(itemToColor, itemColor, toDegree / (startDegree + 360));
114  }
115  UGL.DrawDoughnut(vh, serie.runtimeCenterPos, insideRadius, outsideRadius, itemColor, itemToColor,
116  Color.clear, startDegree, toDegree, borderWidth, borderColor, 0, chart.settings.cicleSmoothness,
117  roundCap, serie.clockwise);
118  DrawCenter(vh, serie, serieData, insideRadius, j == data.Count - 1);
119  UpateLabelPosition(serie, serieData, j, startDegree, toDegree, centerRadius);
120  }
121  if (!serie.animation.IsFinish())
122  {
123  serie.animation.CheckProgress(360);
124  chart.RefreshChart();
125  }
126  if (dataChanging)
127  {
128  chart.RefreshChart();
129  }
130  }

243  private void UpateLabelPosition(Serie serie, SerieData serieData, int index, float startAngle,
244  float toAngle, float centerRadius)
245  {
246  if (!serie.label.show) return;
247  if (serieData.labelObject == null) return;
248  switch (serie.label.position)
249  {
250  case SerieLabel.Position.Center:
251  serieData.labelPosition = serie.runtimeCenterPos + serie.label.offset;
252  break;
253  case SerieLabel.Position.Bottom:
254  var px1 = Mathf.Sin(startAngle * Mathf.Deg2Rad) * centerRadius;
255  var py1 = Mathf.Cos(startAngle * Mathf.Deg2Rad) * centerRadius;
256  var xDiff = serie.clockwise ? -serie.label.margin : serie.label.margin;
257  serieData.labelPosition = serie.runtimeCenterPos + new Vector3(px1 + xDiff, py1);
258  break;
259  case SerieLabel.Position.Top:
260  startAngle += serie.clockwise ? -serie.label.margin : serie.label.margin;
261  toAngle += serie.clockwise ? serie.label.margin : -serie.label.margin;
262  var px2 = Mathf.Sin(toAngle * Mathf.Deg2Rad) * centerRadius;
263  var py2 = Mathf.Cos(toAngle * Mathf.Deg2Rad) * centerRadius;
264  serieData.labelPosition = serie.runtimeCenterPos + new Vector3(px2, py2);
265  break;
266  }
267  serieData.labelObject.SetLabelPosition(serieData.labelPosition);
268  }

320  private int GetRingIndex(Serie serie, Vector2 local)
321  {
322  if (serie.type != SerieType.Ring) return -1;
323  var dist = Vector2.Distance(local, serie.runtimeCenterPos);
324  if (dist > serie.runtimeOutsideRadius) return -1;
325  Vector2 dir = local - new Vector2(serie.runtimeCenterPos.x, serie.runtimeCenterPos.y);
326  float angle = VectorAngle(Vector2.up, dir);
327  for (int i = 0; i < serie.data.Count; i++)
328  {
329  var serieData = serie.data[i];
330  if (dist >= serieData.runtimePieInsideRadius &&
331  dist <= serieData.runtimePieOutsideRadius &&
332  angle >= serieData.runtimePieStartAngle &&
333  angle <= serieData.runtimePieToAngle)
334  {
335  return i;
336  }
337  }
338  return -1;
339  }

351  private float VectorAngle(Vector2 from, Vector2 to)
352  {
353  float angle;
354 
355  Vector3 cross = Vector3.Cross(from, to);
356  angle = Vector2.Angle(from, to);
357  angle = cross.z > 0 ? -angle : angle;
358  angle = (angle + 360) % 360;
359  return angle;
