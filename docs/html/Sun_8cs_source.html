169  public static void CalculateSunPosition(
170  DateTime dateTime, double latitude, double longitude, out double outAzimuth, out double outAltitude)
171  {
172  // Convert to UTC
173  dateTime = dateTime.ToUniversalTime();
174 
175  // Number of days from J2000.0.
176  double julianDate = 367 * dateTime.Year -
177  (int)((7.0 / 4.0) * (dateTime.Year +
178  (int)((dateTime.Month + 9.0) / 12.0))) +
179  (int)((275.0 * dateTime.Month) / 9.0) +
180  dateTime.Day - 730531.5;
181 
182  double julianCenturies = julianDate / 36525.0;
183 
184  // Sidereal Time
185  double siderealTimeHours = 6.6974 + 2400.0513 * julianCenturies;
186 
187  double siderealTimeUT = siderealTimeHours +
188  (366.2422 / 365.2422) * (double)dateTime.TimeOfDay.TotalHours;
189 
190  double siderealTime = siderealTimeUT * 15 + longitude;
191 
192  // Refine to number of days (fractional) to specific time.
193  julianDate += (double)dateTime.TimeOfDay.TotalHours / 24.0;
194  julianCenturies = julianDate / 36525.0;
195 
196  // Solar Coordinates
197  double meanLongitude = CorrectAngle(Deg2Rad *
198  (280.466 + 36000.77 * julianCenturies));
199 
200  double meanAnomaly = CorrectAngle(Deg2Rad *
201  (357.529 + 35999.05 * julianCenturies));
202 
203  double equationOfCenter = Deg2Rad * ((1.915 - 0.005 * julianCenturies) *
204  Math.Sin(meanAnomaly) + 0.02 * Math.Sin(2 * meanAnomaly));
205 
206  double elipticalLongitude =
207  CorrectAngle(meanLongitude + equationOfCenter);
208 
209  double obliquity = (23.439 - 0.013 * julianCenturies) * Deg2Rad;
210 
211  // Right Ascension
212  double rightAscension = Math.Atan2(
213  Math.Cos(obliquity) * Math.Sin(elipticalLongitude),
214  Math.Cos(elipticalLongitude));
215 
216  double declination = Math.Asin(
217  Math.Sin(rightAscension) * Math.Sin(obliquity));
218 
219  // Horizontal Coordinates
220  double hourAngle = CorrectAngle(siderealTime * Deg2Rad) - rightAscension;
221 
222  if (hourAngle > Math.PI)
223  {
224  hourAngle -= 2 * Math.PI;
225  }
226 
227  double altitude = Math.Asin(Math.Sin(latitude * Deg2Rad) *
228  Math.Sin(declination) + Math.Cos(latitude * Deg2Rad) *
229  Math.Cos(declination) * Math.Cos(hourAngle));
230 
231  // Nominator and denominator for calculating Azimuth
232  // angle. Needed to test which quadrant the angle is in.
233  double aziNom = -Math.Sin(hourAngle);
234  double aziDenom =
235  Math.Tan(declination) * Math.Cos(latitude * Deg2Rad) -
236  Math.Sin(latitude * Deg2Rad) * Math.Cos(hourAngle);
237 
238  double azimuth = Math.Atan(aziNom / aziDenom);
239 
240  if (aziDenom < 0) // In 2nd or 3rd quadrant
241  {
242  azimuth += Math.PI;
243  }
244  else if (aziNom < 0) // In 4th quadrant
245  {
246  azimuth += 2 * Math.PI;
247  }
248 
249  outAltitude = altitude;
250  outAzimuth = azimuth;
251  }
