92  public static void GetBezierList(ref List<Vector3> posList, Vector3 sp, Vector3 ep,
93  Vector3 lsp, Vector3 nep, float smoothness = 2f, float k = 2.0f)
94  {
95  float dist = Mathf.Abs(sp.x - ep.x);
96  Vector3 cp1, cp2;
97  var dir = (ep - sp).normalized;
98  var diff = dist / k;
99  if (lsp == sp)
100  {
101  cp1 = sp + dist / k * dir * 1;
102  cp1.y = sp.y;
103  cp1 = sp;
104  }
105  else
106  {
107  cp1 = sp + (ep - lsp).normalized * diff;
108  }
109  if (nep == ep) cp2 = ep;
110  else cp2 = ep - (nep - sp).normalized * diff;
111  dist = Vector3.Distance(sp, ep);
112  int segment = (int)(dist / (smoothness <= 0 ? 2f : smoothness));
113  if (segment < 1) segment = (int)(dist / 0.5f);
114  if (segment < 4) segment = 4;
115  GetBezierList2(ref posList, sp, ep, segment, cp1, cp2);
116  }

164  public static Vector3 GetBezier2(float t, Vector3 sp, Vector3 p1, Vector3 p2, Vector3 ep)
165  {
166  t = Mathf.Clamp01(t);
167  var oneMinusT = 1f - t;
168  return oneMinusT * oneMinusT * oneMinusT * sp +
