167  void Update()
168  {
169  // Exit Sample
170 
171  if (IsEscapePressed())

179  // Hide and lock cursor when right mouse button pressed
180  if (IsRightMouseButtonDown())
181  {
182  Cursor.lockState = CursorLockMode.Locked;
183  }
184 
185  // Unlock and show cursor when right mouse button released
186  if (IsRightMouseButtonUp())
187  {
188  Cursor.visible = true;
189  Cursor.lockState = CursorLockMode.None;
190  }
191 
192  // Rotation
193  if (IsCameraRotationAllowed())
194  {
195  var mouseMovement = GetInputLookRotation() * k_MouseSensitivityMultiplier * mouseSensitivity;
196  if (invertY)
197  mouseMovement.y = -mouseMovement.y;
198 
199  var mouseSensitivityFactor = mouseSensitivityCurve.Evaluate(mouseMovement.magnitude);
200 
201  m_TargetCameraState.yaw += mouseMovement.x * mouseSensitivityFactor;
202  m_TargetCameraState.pitch += mouseMovement.y * mouseSensitivityFactor;
203  }
204 
205  // Translation
206  var translation = GetInputTranslationDirection() * Time.deltaTime;
207 
208  // Speed up movement when shift key held
209  if (IsBoostPressed())
210  {
211  translation *= 10.0f;
212  }
213 
214  // Modify movement by a boost factor (defined in Inspector and modified in play mode through the mouse scroll wheel)
215  boost += GetBoostFactor();
216  translation *= Mathf.Pow(2.0f, boost);
217 
218  m_TargetCameraState.Translate(translation);
219 
220  // Framerate-independent interpolation
221  // Calculate the lerp amount, such that we get 99% of the way to our target in the specified time
222  var positionLerpPct = 1f - Mathf.Exp((Mathf.Log(1f - 0.99f) / positionLerpTime) * Time.deltaTime);
223  var rotationLerpPct = 1f - Mathf.Exp((Mathf.Log(1f - 0.99f) / rotationLerpTime) * Time.deltaTime);
224  m_InterpolatingCameraState.LerpTowards(m_TargetCameraState, positionLerpPct, rotationLerpPct);
225 
226  m_InterpolatingCameraState.UpdateTransform(transform);
227  }
