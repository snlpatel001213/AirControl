166  private void UpdateRuntimeData(Serie serie)
167  {
168  var data = serie.data;
169  serie.runtimeDataMax = serie.yMax;
170  serie.runtimePieDataTotal = serie.yTotal;
171 
172  SerieHelper.UpdateCenter(serie, chart.chartPosition, chart.chartWidth, chart.chartHeight);
173  float totalDegree = 0;
174  float startDegree = 0;
175  float zeroReplaceValue = 0;
176  int showdataCount = 0;
177  foreach (var sd in serie.data)
178  {
179  if (sd.show && serie.pieRoseType == RoseType.Area) showdataCount++;
180  sd.canShowLabel = false;
181  }
182  float dataChangeDuration = serie.animation.GetUpdateAnimationDuration();
183  bool isAllZeroValue = SerieHelper.IsAllZeroValue(serie, 1);
184  var dataTotalFilterMinAngle = serie.runtimePieDataTotal;
185  if (isAllZeroValue)
186  {
187  totalDegree = 360;
188  zeroReplaceValue = totalDegree / data.Count;
189  serie.runtimeDataMax = zeroReplaceValue;
190  serie.runtimePieDataTotal = 360;
191  dataTotalFilterMinAngle = 360;
192  }
193  else
194  {
195  dataTotalFilterMinAngle = GetTotalAngle(serie, serie.runtimePieDataTotal, ref totalDegree);
196  }
197  serie.animation.InitProgress(data.Count, 0, 360);
198  for (int n = 0; n < data.Count; n++)
199  {
200  var serieData = data[n];
201  serieData.index = n;
202  var value = isAllZeroValue ? zeroReplaceValue : serieData.GetCurrData(1, dataChangeDuration);
203  serieData.runtimePieStartAngle = startDegree;
204  serieData.runtimePieToAngle = startDegree;
205  serieData.runtimePieHalfAngle = startDegree;
206  serieData.runtimePieCurrAngle = startDegree;
207  if (!serieData.show)
208  {
209  continue;
210  }
211  float degree = serie.pieRoseType == RoseType.Area
212  ? (totalDegree / showdataCount)
213  : (float)(totalDegree * value / dataTotalFilterMinAngle);
214  if (serie.minAngle > 0 && degree < serie.minAngle) degree = serie.minAngle;
215  serieData.runtimePieToAngle = startDegree + degree;
216  serieData.runtimePieOutsideRadius = serie.pieRoseType > 0 ?
217  serie.runtimeInsideRadius + (float)((serie.runtimeOutsideRadius - serie.runtimeInsideRadius) * value / serie.runtimeDataMax) :
218  serie.runtimeOutsideRadius;
219  if (serieData.highlighted)
220  {
221  serieData.runtimePieOutsideRadius += chart.theme.serie.pieTooltipExtraRadius;
222  }
223  var offset = 0f;
224  if (serie.pieClickOffset && serieData.selected)
225  {
226  offset += chart.theme.serie.pieSelectedOffset;
227  }
228  if (serie.animation.CheckDetailBreak(serieData.runtimePieToAngle))
229  {
230  serieData.runtimePieCurrAngle = serie.animation.GetCurrDetail();
231  }
232  else
233  {
234  serieData.runtimePieCurrAngle = serieData.runtimePieToAngle;
235  }
236  var halfDegree = (serieData.runtimePieToAngle - startDegree) / 2;
237  serieData.runtimePieHalfAngle = startDegree + halfDegree;
238  serieData.runtiemPieOffsetCenter = serie.runtimeCenterPos;
239  serieData.runtimePieInsideRadius = serie.runtimeInsideRadius;
240  if (offset > 0)
241  {
242  var currRad = serieData.runtimePieHalfAngle * Mathf.Deg2Rad;
243  var currSin = Mathf.Sin(currRad);
244  var currCos = Mathf.Cos(currRad);
245  serieData.runtimePieOffsetRadius = 0;
246  serieData.runtimePieInsideRadius -= serieData.runtimePieOffsetRadius;
247  serieData.runtimePieOutsideRadius -= serieData.runtimePieOffsetRadius;
248  if (serie.pieClickOffset && serieData.selected)
249  {
250  serieData.runtimePieOffsetRadius += chart.theme.serie.pieSelectedOffset;
251  if (serieData.runtimePieInsideRadius > 0)
252  {
253  serieData.runtimePieInsideRadius += chart.theme.serie.pieSelectedOffset;
254  }
255  serieData.runtimePieOutsideRadius += chart.theme.serie.pieSelectedOffset;
256  }
257  serieData.runtiemPieOffsetCenter = new Vector3(
258  serie.runtimeCenterPos.x + serieData.runtimePieOffsetRadius * currSin,
259  serie.runtimeCenterPos.y + serieData.runtimePieOffsetRadius * currCos);
260  }
261  serieData.canShowLabel = serieData.runtimePieCurrAngle >= serieData.runtimePieHalfAngle;
262  startDegree = serieData.runtimePieToAngle;
263  SerieLabelHelper.UpdatePieLabelPosition(serie, serieData);
264  }
265  SerieLabelHelper.AvoidLabelOverlap(serie, chart.theme.common);
266  }

268  private double GetTotalAngle(Serie serie, double dataTotal, ref float totalAngle)
269  {
270  totalAngle = 360f;
271  if (serie.minAngle > 0)
272  {
273  var rate = serie.minAngle / 360;
274  var minAngleValue = dataTotal * rate;
275  foreach (var serieData in serie.data)
276  {
277  var value = serieData.GetData(1);
278  if (value < minAngleValue)
279  {
280  totalAngle -= serie.minAngle;
281  dataTotal -= value;
282  }
283  }
284  return dataTotal;
285  }
286  else
287  {
288  return dataTotal;
289  }
290  }

301  private void DrawPie(VertexHelper vh, Serie serie)
302  {
303  var data = serie.data;
304  serie.animation.InitProgress(data.Count, 0, 360);
305  if (!serie.show || serie.animation.HasFadeOut())
306  {
307  return;
308  }
309  bool dataChanging = false;
310  for (int n = 0; n < data.Count; n++)
311  {
312  var serieData = data[n];
313  if (!serieData.show)
314  {
315  continue;
316  }
317  var itemStyle = SerieHelper.GetItemStyle(serie, serieData, serieData.highlighted);
318  if (serieData.IsDataChanged()) dataChanging = true;
319  var serieNameCount = chart.m_LegendRealShowName.IndexOf(serieData.legendName);
320  var color = SerieHelper.GetItemColor(serie, serieData, chart.theme, serieNameCount,
321  serieData.highlighted);
322  var toColor = SerieHelper.GetItemToColor(serie, serieData, chart.theme, serieNameCount,
323  serieData.highlighted);
324  var borderWidth = itemStyle.borderWidth;
325  var borderColor = itemStyle.borderColor;
326 
327  if (serie.pieClickOffset && serieData.selected)
328  {
329  var drawEndDegree = serieData.runtimePieCurrAngle;
330  var needRoundCap = serie.roundCap && serieData.runtimePieInsideRadius > 0;
331  UGL.DrawDoughnut(vh, serieData.runtiemPieOffsetCenter, serieData.runtimePieInsideRadius,
332  serieData.runtimePieOutsideRadius, color, toColor, Color.clear, serieData.runtimePieStartAngle,
333  drawEndDegree, borderWidth, borderColor, serie.pieSpace / 2, chart.settings.cicleSmoothness,
334  needRoundCap, true);
335  }
336  else
337  {
338  var drawEndDegree = serieData.runtimePieCurrAngle;
339  var needRoundCap = serie.roundCap && serieData.runtimePieInsideRadius > 0;
340  UGL.DrawDoughnut(vh, serie.runtimeCenterPos, serieData.runtimePieInsideRadius,
341  serieData.runtimePieOutsideRadius, color, toColor, Color.clear, serieData.runtimePieStartAngle,
342  drawEndDegree, borderWidth, borderColor, serie.pieSpace / 2, chart.settings.cicleSmoothness,
343  needRoundCap, true);
344  DrawPieCenter(vh, serie, itemStyle, serieData.runtimePieInsideRadius);
345  }
346  if (!serie.animation.CheckDetailBreak(serieData.runtimePieToAngle)) serie.animation.SetDataFinish(n);
347  else break;
348  }
349  if (!serie.animation.IsFinish())
350  {
351  serie.animation.CheckProgress(360);
352  serie.animation.CheckSymbol(serie.symbol.GetSize(null, chart.theme.serie.lineSymbolSize));
353  chart.RefreshPainter(serie);
354  }
355  if (dataChanging)
356  {
357  chart.RefreshPainter(serie);
358  }
359  chart.raycastTarget = IsAnyPieClickOffset() || IsAnyPieDataHighlight();
360  }

414  private void DrawPieLabelLine(VertexHelper vh, Serie serie, SerieData serieData, Color color)
415  {
416  if (serie.animation.enable && serie.animation.HasFadeOut()) return;
417  var serieLabel = SerieHelper.GetSerieLabel(serie, serieData);
418  if (serieLabel.show
419  && serieLabel.position == SerieLabel.Position.Outside
420  && serieLabel.line)
421  {
422  var insideRadius = serieData.runtimePieInsideRadius;
423  var outSideRadius = serieData.runtimePieOutsideRadius;
424  var center = serie.runtimeCenterPos;
425  var currAngle = serieData.runtimePieHalfAngle;
426  if (!ChartHelper.IsClearColor(serieLabel.lineColor)) color = serieLabel.lineColor;
427  else if (serieLabel.lineType == SerieLabel.LineType.HorizontalLine) color *= color;
428  float currSin = Mathf.Sin(currAngle * Mathf.Deg2Rad);
429  float currCos = Mathf.Cos(currAngle * Mathf.Deg2Rad);
430  var radius1 = serieLabel.lineType == SerieLabel.LineType.HorizontalLine ?
431  serie.runtimeOutsideRadius : outSideRadius;
432  var radius2 = serie.runtimeOutsideRadius + serieLabel.lineLength1;
433  var radius3 = insideRadius + (outSideRadius - insideRadius) / 2;
434  if (radius1 < serie.runtimeInsideRadius) radius1 = serie.runtimeInsideRadius;
435  radius1 -= 0.1f;
436  var pos0 = new Vector3(center.x + radius3 * currSin, center.y + radius3 * currCos);
437  var pos1 = new Vector3(center.x + radius1 * currSin, center.y + radius1 * currCos);
438  var pos2 = serieData.labelPosition;
439  if (pos2.x == 0)
440  {
441  pos2 = new Vector3(center.x + radius2 * currSin, center.y + radius2 * currCos);
442  }
443  Vector3 pos4, pos6;
444  var horizontalLineCircleRadius = serieLabel.lineWidth * 4f;
445  var lineCircleDiff = horizontalLineCircleRadius - 0.3f;
446  if (currAngle < 90)
447  {
448  var r4 = Mathf.Sqrt(radius1 * radius1 - Mathf.Pow(currCos * radius3, 2)) - currSin * radius3;
449  r4 += serieLabel.lineLength1 - lineCircleDiff;
450  pos6 = pos0 + Vector3.right * lineCircleDiff;
451  pos4 = pos6 + Vector3.right * r4;
452  }
453  else if (currAngle < 180)
454  {
455  var r4 = Mathf.Sqrt(radius1 * radius1 - Mathf.Pow(currCos * radius3, 2)) - currSin * radius3;
456  r4 += serieLabel.lineLength1 - lineCircleDiff;
457  pos6 = pos0 + Vector3.right * lineCircleDiff;
458  pos4 = pos6 + Vector3.right * r4;
459  }
460  else if (currAngle < 270)
461  {
462  var currSin1 = Mathf.Sin((360 - currAngle) * Mathf.Deg2Rad);
463  var currCos1 = Mathf.Cos((360 - currAngle) * Mathf.Deg2Rad);
464  var r4 = Mathf.Sqrt(radius1 * radius1 - Mathf.Pow(currCos1 * radius3, 2)) - currSin1 * radius3;
465  r4 += serieLabel.lineLength1 - lineCircleDiff;
466  pos6 = pos0 + Vector3.left * lineCircleDiff;
467  pos4 = pos6 + Vector3.left * r4;
468  }
469  else
470  {
471  var currSin1 = Mathf.Sin((360 - currAngle) * Mathf.Deg2Rad);
472  var currCos1 = Mathf.Cos((360 - currAngle) * Mathf.Deg2Rad);
473  var r4 = Mathf.Sqrt(radius1 * radius1 - Mathf.Pow(currCos1 * radius3, 2)) - currSin1 * radius3;
474  r4 += serieLabel.lineLength1 - lineCircleDiff;
475  pos6 = pos0 + Vector3.left * lineCircleDiff;
476  pos4 = pos6 + Vector3.left * r4;
477  }
478  var pos5X = currAngle > 180 ? pos2.x - serieLabel.lineLength2 : pos2.x + serieLabel.lineLength2;
479  var pos5 = new Vector3(pos5X, pos2.y);
480  switch (serieLabel.lineType)
481  {
482  case SerieLabel.LineType.BrokenLine:
483  UGL.DrawLine(vh, pos1, pos2, pos5, serieLabel.lineWidth, color);
484  break;
485  case SerieLabel.LineType.Curves:
486  UGL.DrawCurves(vh, pos1, pos5, pos1, pos2, serieLabel.lineWidth, color,
487  chart.settings.lineSmoothness);
488  break;
489  case SerieLabel.LineType.HorizontalLine:
490  UGL.DrawCricle(vh, pos0, horizontalLineCircleRadius, color);
491  UGL.DrawLine(vh, pos6, pos4, serieLabel.lineWidth, color);
492  break;
493  }
494  }
495  }

497  private void DrawPieLabel(Serie serie, int dataIndex, SerieData serieData, Color serieColor)
498  {
499  if (serieData.labelObject == null) return;
500  if (serie.animation.enable && serie.animation.HasFadeOut()) return;
501  var currAngle = serieData.runtimePieHalfAngle;
502  var isHighlight = (serieData.highlighted && serie.emphasis.label.show);
503  var serieLabel = SerieHelper.GetSerieLabel(serie, serieData);
504  var iconStyle = SerieHelper.GetIconStyle(serie, serieData);
505  var showLabel = ((serieLabel.show || isHighlight) && serieData.canShowLabel);
506  if (showLabel)
507  {
508  serieData.SetLabelActive(showLabel);
509  float rotate = 0;
510  bool isInsidePosition = serieLabel.position == SerieLabel.Position.Inside;
511  if (serieLabel.textStyle.rotate > 0 && isInsidePosition)
512  {
513  if (currAngle > 180) rotate += 270 - currAngle;
514  else rotate += -(currAngle - 90);
515  }
516  Color color = serieColor;
517  if (isHighlight)
518  {
519  if (!ChartHelper.IsClearColor(serie.emphasis.label.textStyle.color))
520  {
521  color = serie.emphasis.label.textStyle.color;
522  }
523  }
524  else if (!ChartHelper.IsClearColor(serieLabel.textStyle.color))
525  {
526  color = serieLabel.textStyle.color;
527  }
528  else
529  {
530  color = isInsidePosition ? Color.white : serieColor;
531  }
532  var fontSize = isHighlight
533  ? serie.emphasis.label.textStyle.GetFontSize(chart.theme.common)
534  : serieLabel.textStyle.GetFontSize(chart.theme.common);
535  var fontStyle = isHighlight
536  ? serie.emphasis.label.textStyle.fontStyle
537  : serieLabel.textStyle.fontStyle;
538 
539  serieData.labelObject.label.SetColor(color);
540  serieData.labelObject.label.SetFontSize(fontSize);
541  serieData.labelObject.label.SetFontStyle(fontStyle);
542  serieData.labelObject.SetLabelRotate(rotate);
543  if (!string.IsNullOrEmpty(serieLabel.formatter))
544  {
545  var value = serieData.data[1];
546  var total = serie.yTotal;
547  var content = SerieLabelHelper.GetFormatterContent(serie, serieData, value, total,
548  serieLabel, serieColor);
549  if (serieData.labelObject.SetText(content)) chart.RefreshPainter(serie);
550  }
551  else
552  {
553  if (serieData.labelObject.SetText(serieData.name)) chart.RefreshPainter(serie);
554  }
555  serieData.labelObject.SetPosition(SerieLabelHelper.GetRealLabelPosition(serieData, serieLabel));
556  if (showLabel) serieData.labelObject.SetLabelPosition(serieLabel.offset);
557  else serieData.SetLabelActive(false);
558  }
559  else
560  {
561  serieData.SetLabelActive(false);
562  }
563  serieData.labelObject.UpdateIcon(iconStyle);
564  }

566  protected int GetPiePosIndex(Serie serie, Vector2 local)
567  {
568  if (serie.type != SerieType.Pie) return -1;
569  var dist = Vector2.Distance(local, serie.runtimeCenterPos);
570  var maxRadius = serie.runtimeOutsideRadius + 3 * chart.theme.serie.pieSelectedOffset;
571  if (dist < serie.runtimeInsideRadius || dist > maxRadius) return -1;
572  Vector2 dir = local - new Vector2(serie.runtimeCenterPos.x, serie.runtimeCenterPos.y);
573  float angle = ChartHelper.GetAngle360(Vector2.up, dir);
574  for (int i = 0; i < serie.data.Count; i++)
575  {
576  var serieData = serie.data[i];
577  if (angle >= serieData.runtimePieStartAngle && angle <= serieData.runtimePieToAngle)
578  {
579  var ndist = !serieData.selected ? dist :
580  Vector2.Distance(local, serieData.runtiemPieOffsetCenter);
581  if (ndist >= serieData.runtimePieInsideRadius && ndist <= serieData.runtimePieOutsideRadius)
582  {
583  return i;
584  }
585  }
586  }
587  return -1;
