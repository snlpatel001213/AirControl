34  public static string FloatToStr(double value, string numericFormatter = "F", int precision = 0)
35  {
36  if (precision > 0 && numericFormatter.Length == 1)
37  {
38  if (!s_PrecisionToStr.ContainsKey(precision))
39  {
40  s_PrecisionToStr[precision] = new Dictionary<string, string>();
41  }
42  if (!s_PrecisionToStr[precision].ContainsKey(numericFormatter))
43  {
44  s_PrecisionToStr[precision][numericFormatter] = numericFormatter + precision;
45  }
46  return NumberToStr(value, s_PrecisionToStr[precision][numericFormatter]);
47  }
48  else

54  public static string NumberToStr(double value, string formatter)
55  {
56  if (!s_NumberToStr.ContainsKey(value))
57  {
58  s_NumberToStr[value] = new Dictionary<string, string>();
59  }
60  if (!s_NumberToStr[value].ContainsKey(formatter))
61  {
62  if (string.IsNullOrEmpty(formatter))
63  {
64  if (value - (int)value == 0)
65  s_NumberToStr[value][formatter] = ((int)value).ToString();
66  else
67  s_NumberToStr[value][formatter] = value.ToString();
68  }
69  else if (formatter.StartsWith(NUMERIC_FORMATTER_D)
70  || formatter.StartsWith(NUMERIC_FORMATTER_d)
71  || formatter.StartsWith(NUMERIC_FORMATTER_X)
72  || formatter.StartsWith(NUMERIC_FORMATTER_x)
73  )
74  {
75  s_NumberToStr[value][formatter] = ((int)value).ToString(formatter, ci);
76  }
77  else
78  {
79  s_NumberToStr[value][formatter] = value.ToString(formatter, ci);
80  }
81  }
82  return s_NumberToStr[value][formatter];
83  }
