119  private static void CheckSerie(BaseChart chart, StringBuilder sb)
120  {
121  var allDataIsEmpty = true;
122  var allDataIsZero = true;
123  var allSerieIsHide = true;
124  foreach (var serie in chart.series.list)
125  {
126  if (serie.show) allSerieIsHide = false;
127  if (serie.dataCount > 0)
128  {
129  allDataIsEmpty = false;
130  for (int i = 0; i < serie.dataCount; i++)
131  {
132  var serieData = serie.GetSerieData(i);
133  for (int j = 1; j < serieData.data.Count; j++)
134  {
135  if (serieData.GetData(j) != 0)
136  {
137  allDataIsZero = false;
138  break;
139  }
140  }
141  }
142  var dataCount = serie.GetSerieData(0).data.Count;
143  if (serie.showDataDimension > 1 && serie.showDataDimension != dataCount)
144  {
145  sb.AppendFormat("warning:serie {0} serieData.data.count[{1}] not match showDataDimension[{2}]\n", serie.index, dataCount, serie.showDataDimension);
146  }
147  }
148  else
149  {
150  sb.AppendFormat("warning:serie {0} no data\n", serie.index);
151  }
152  if (IsColorAlphaZero(serie.itemStyle.color))
153  sb.AppendFormat("warning:serie {0} itemStyle->color alpha is 0\n", serie.index);
154  if (serie.itemStyle.opacity == 0)
155  sb.AppendFormat("warning:serie {0} itemStyle->opacity is 0\n", serie.index);
156  if (serie.itemStyle.borderWidth != 0 && IsColorAlphaZero(serie.itemStyle.borderColor))
157  sb.AppendFormat("warning:serie {0} itemStyle->borderColor alpha is 0\n", serie.index);
158  switch (serie.type)
159  {
160  case SerieType.Line:
161  if (serie.lineStyle.opacity == 0)
162  sb.AppendFormat("warning:serie {0} lineStyle->opacity is 0\n", serie.index);
163  if (IsColorAlphaZero(serie.lineStyle.color))
164  sb.AppendFormat("warning:serie {0} lineStyle->color alpha is 0\n", serie.index);
165  break;
166  case SerieType.Bar:
167  if (serie.barWidth == 0)
168  sb.AppendFormat("warning:serie {0} barWidth is 0\n", serie.index);
169  break;
170  case SerieType.Pie:
171  if (serie.radius.Length >= 2 && serie.radius[1] == 0)
172  sb.AppendFormat("warning:serie {0} radius[1] is 0\n", serie.index);
173  break;
174  case SerieType.Scatter:
175  case SerieType.EffectScatter:
176  if (!serie.symbol.show)
177  sb.AppendFormat("warning:serie {0} symbol type is None\n", serie.index);
178  break;
179  }
180  }
181  if (allDataIsEmpty) sb.Append("warning:all serie data is empty\n");
182  if (!allDataIsEmpty && allDataIsZero) sb.Append("warning:all serie data is 0\n");
183  if (allSerieIsHide) sb.Append("warning:all serie is hide\n");
184  }
185  }
186 }
